int invert_Arr(int arr[])
{
	int temp;
	for (int i = 0; i < Size; i++)
	{
		for (int j = 9; j >= 0; j--)
		{
			if (i + j == 9 && i < Size/2)
			{
				temp = arr[i];
				arr[i] = arr[j];
				arr[j] = temp;
			}
		}
	}

	return 0;
}
